get_property(dialect_libs GLOBAL PROPERTY MLIR_DIALECT_LIBS)
get_property(conversion_libs GLOBAL PROPERTY MLIR_CONVERSION_LIBS)

# Link order matters for static libraries!
# Libraries that are depended upon must come AFTER the libraries that depend on them.
# AscendModelTransforms depends on AscendModelAnalysis, so Analysis comes after Transforms.
# We list Analysis twice to handle potential circular dependencies.
set(LIBS
  ${dialect_libs}
  ${conversion_libs}
  AscendModelTransforms
  AscendModelAnalysis
  AscendModelIR
  AscendModelAnalysis  # Listed again for circular dependency resolution
  MLIROptLib
)

add_llvm_executable(tritonsim-opt
  tritonsim-opt.cpp
)

llvm_update_compile_flags(tritonsim-opt)
target_link_libraries(tritonsim-opt PRIVATE ${LIBS})

# Conditionally link Triton dialect library
if(TRITONSIM_HAS_TRITON AND TRITON_DIALECT_LIB)
  target_link_libraries(tritonsim-opt PRIVATE ${TRITON_DIALECT_LIB})
endif()

mlir_check_all_link_libraries(tritonsim-opt)
